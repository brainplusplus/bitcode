dan query builder yao :

import { Process } from "@yao/runtime";

function LatestOrders() {
  const orders = Process(
    "models.order.Get",
    { wheres: [{ column: "status", value: "paid" }] }
  );
  return orders.map(o => ({
    id: o.id,
    customer: o.customer,
    total: "$" + o.total
  }));
}
