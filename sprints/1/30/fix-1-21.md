// String
"string"                → VARCHAR(255)
"string:uuid"           → UUID / CHAR(36)
"string:ip"             → INET / VARCHAR(45)
"string:mac"            → MACADDR / VARCHAR(17)

// Text
"text"                  → TEXT
"text:medium"           → MEDIUMTEXT / TEXT
"text:long"             → LONGTEXT / TEXT

// Integer
"integer"               → INTEGER
"integer:tiny"          → TINYINT
"integer:small"         → SMALLINT
"integer:medium"        → MEDIUMINT / INTEGER
"integer:big"           → BIGINT

// Decimal / Numeric
"decimal"               → DECIMAL(18,2)
"numeric"               → NUMERIC (PostgreSQL unlimited) / DECIMAL(65,30)

// JSON
"json"                  → JSONB / JSON / TEXT (no default, any shape)
"json:object"           → JSONB DEFAULT '{}' / JSON DEFAULT (JSON_OBJECT())
"json:array"            → JSONB DEFAULT '[]' / JSON DEFAULT (JSON_ARRAY())

// DateTime
"datetime"              → DATETIME / TIMESTAMPTZ
"datetime:tz"           → TIMESTAMPTZ (explicit timezone)
"timestamp"             → alias for datetime
"timestamp:tz"          → alias for datetime:tz

// Binary
"binary"                → BYTEA / BLOB

// Vector
"vector"                → vector(dimensions) / JSON
